顶部菜单 ​

vue
<template>
  <el-menu
    :default-active="activeMenu"
    mode="horizontal"
    @select="handleSelect"
  >
    <template v-for="(item, index) in topMenus">
      <el-menu-item :style="{'--theme': theme}" :index="item.path" :key="index" v-if="index < visibleNumber">
        <svg-icon :icon-class="item.meta.icon"/>
        {{ item.meta.title }}
      </el-menu-item>
    </template>


    <!-- 顶部菜单超出数量折叠 -->
    <el-submenu :style="{'--theme': theme}" index="more" v-if="topMenus.length > visibleNumber">
      <template slot="title">更多菜单</template>
      <template v-for="(item, index) in topMenus">
        <el-menu-item
          :index="item.path"
          :key="index"
          v-if="index >= visibleNumber"
        >
          <svg-icon :icon-class="item.meta.icon"/>
          {{ item.meta.title }}
        </el-menu-item>
      </template>
    </el-submenu>
  </el-menu>
</template>


<script>
import { constantRoutes } from "@/router";


// 隐藏侧边栏路由
const hideList = ['/index', '/user/profile'];


export default {
  data() {
    return {
      // 顶部栏初始数
      visibleNumber: 5,
      // 当前激活菜单的 index
      currentIndex: undefined
    };
  },
  computed: {
    theme() {
      return this.$store.state.settings.theme;
    },
    // 顶部显示菜单
    topMenus() {
      let topMenus = [];
      this.routers.map((menu) => {
        if (menu.hidden !== true) {
          // 兼容顶部栏一级菜单内部跳转
          if (menu.path === "/") {
            topMenus.push(menu.children[0]);
          } else {
            topMenus.push(menu);
          }
        }
      });
      return topMenus;
    },
    // 所有的路由信息
    routers() {
      return this.$store.state.permission.topbarRouters;
    },
    // 设置子路由
    childrenMenus() {
      const childrenMenus = [];
      this.routers.map((router) => {
        for (let item in router.children) {
          if (router.children[item].parentPath === undefined) {
            if(router.path === "/") {
              router.children[item].path = "/" + router.children[item].path;
            } else {
              if(!this.ishttp(router.children[item].path)) {
                router.children[item].path = router.path + "/" + router.children[item].path;
              }
            }
            router.children[item].parentPath = router.path;
          }
          childrenMenus.push(router.children[item]);
        }
      });
      return constantRoutes.concat(childrenMenus);
    },
    // 默认激活的菜单
    activeMenu() {
      const path = this.$route.path;
      let activePath = path;
      if (path !== undefined && path.lastIndexOf("/") > 0 && hideList.indexOf(path) === -1) {
        const tmpPath = path.substring(1, path.length);
        activePath = "/" + tmpPath.substring(0, tmpPath.indexOf("/"));
        if (!this.$route.meta.link) {
          this.$store.dispatch('app/toggleSideBarHide', false)
        }
      } else if(!this.$route.children) {
        activePath = path;
        this.$store.dispatch('app/toggleSideBarHide', true);
      }
      this.activeRoutes(activePath);
      return activePath;
    },
  },
  beforeMount() {
    window.addEventListener('resize', this.setVisibleNumber)
  },
  beforeDestroy() {
    window.removeEventListener('resize', this.setVisibleNumber)
  },
  mounted() {
    this.setVisibleNumber();
  },
  methods: {
    // 根据宽度计算设置显示栏数
    setVisibleNumber() {
      const width = document.body.getBoundingClientRect().width / 3;
      this.visibleNumber = parseInt(width / 85);
    },
    // 菜单选择事件
    handleSelect(key, keyPath) {
      this.currentIndex = key;
      const route = this.routers.find(item => item.path === key);
      if (this.ishttp(key)) {
        // http(s):// 路径新窗口打开
        window.open(key, "_blank");
      } else if (!route || !route.children) {
        // 没有子路由路径内部打开
        this.$router.push({ path: key });
        this.$store.dispatch('app/toggleSideBarHide', true);
      } else {
        // 显示左侧联动菜单
        this.activeRoutes(key);
        if (!this.$route.meta.link) {
          this.$store.dispatch('app/toggleSideBarHide', false);
        }
      }
    },
    // 当前激活的路由
    activeRoutes(key) {
      const routes = []
      if (this.childrenMenus && this.childrenMenus.length > 0) {
        this.childrenMenus.map((item) => {
          if (key === item.parentPath || (key === "index" && "" === item.path)) {
            routes.push(item);
          }
        });
      }
      if(routes.length > 0) {
        this.$store.commit("SET_SIDEBAR_ROUTERS", routes);
      } else {
        this.$store.dispatch('app/toggleSideBarHide', true);
      }
    },
    ishttp(url) {
      return url.indexOf('http://') !== -1 || url.indexOf('https://') !== -1
    }
  },
};
</script>

    this.title = "修改文档";

/** 提交按钮 */
submitForm() {
  this.$refs["form"].validate(async (valid) => {

    // 判断表单验证结果
    if (!valid) {
      return;
    }

    // 将文档附件上传到服务器
    await this.submitUpload();

    // 修改的提交
    if (this.form.id != null) {
      try {
        // 将文档附件上传结果添加到form
        this.form.attachPathList.push(...this.documentAttachments)
        const response = await updateDocument(this.form)
        this.$modal.msgSuccess("修改成功");
        this.open = false;
        this.getList();
      }catch (e) {
        this.$message.error(e);
      }
      return;
    }
    // 添加的提交
    try {
      // 将文档附件上传结果添加到form
      this.$set(this.form,"files",this.documentAttachments);
      const response = await createDocument(this.form)
      this.$modal.msgSuccess("新增成功");
      this.open = false;
      this.getList();
    }catch (e) {
      this.$message.error(e);
    }

    // 清空上传后返回的文件数组
    this.documentAttachments = []
  });

// 将文档附件上传到服务器
async submitUpload() {
  //判断是否有文件再上传
  if (this.fileList.length === 0) {
    return this.$message.warning('请选取文件后再上传')
  }
  // 下面的代码将创建一个空的FormData对象:
  const formData = new FormData()

  // 有新增的才调用导入文档附件接口
  if (this.fileList.some(item=>item.status===1)){

    this.fileList.forEach((file) => {
      if (file.status===1){
        // 你可以使用FormData.append来添加键/值对到表单里面；
        formData.append('files', file.raw)
      }
    })

    // 添加自定义参数，不传可删除
    // formData.append('parentId', '49')

    // 导入文档附件
    try {
      const res = await importingDocumentAttachments(formData)
      this.$message.success('上传成功');
      this.documentAttachments.push(...res.data);
      this.documentAttachments.forEach(item => item.status = 1);
    }catch (e){
      this.$message.error('上传失败');
    }
  }

  //清空fileList
  this.fileList = []
},

        this.productArea = _.cloneDeep(this.form.productArea);
        this.productSeries = _.cloneDeep(this.form.productSeries);
        this.attachPathList = _.cloneDeep(this.form.attachPathList);

        this.form.documentType = this.form.documentType;
        this.form.productArea = this.form.productArea;
        this.form.productSeries = this.form.productSeries;

        // 修改的提交
        if (this.form.id != null) {
          try {
            // 将文档附件上传结果添加到form
            this.form.attachPathList.push(...this.documentAttachments)
            const response = await updateDocument(this.form)
            this.$modal.msgSuccess("修改成功");
            this.open = false;
            this.getList();
          }catch (e) {
            // this.$message.error(e);
            console.log(111)
            this.form.documentType = this.documentType;
            this.form.productArea = this.productArea;
            this.form.productSeries = this.productSeries;
            this.form.attachPathList = this.attachPathList
            this.documentAttachments = []
            this.dialogLoading = false;
            console.log(this.form,"this.form")
            return;
          }
          this.dialogLoading = false;
          return;
        }

        // 添加的提交
        try {
          // 将文档附件上传结果添加到form
          this.$set(this.form,"files",this.documentAttachments);
          const response = await createDocument(this.form)
          this.$modal.msgSuccess("新增成功");
          this.open = false;
          this.getList();
        }catch (e) {
          // this.$message.error(e);
          this.form.documentType = this.documentType;
          this.form.productArea = this.productArea;
          this.form.productSeries = this.productSeries;
          this.form.attachPathList = this.attachPathList
          this.documentAttachments = []
          this.dialogLoading = false;
          return;
        }

        //清空fileList
        this.fileList = []
        // 清空上传后返回的文件数组
        this.documentAttachments = []
        this.dialogLoading = false;
      });
    },
    /** 删除按钮操作 */
    handleDelete(row) {
      const id = row.id;
      this.$modal.confirm('是否确认删除文档编号为"' + row.documentNumber + '"的数据项?').then(function() {
        return deleteDocument(id);
      }).then(() => {
        this.getList();
        this.$modal.msgSuccess("删除成功");
      }).catch(() => {});
    },
    /** 导出按钮操作 */
    handleExport() {
      // 处理查询参数
      let params = {...this.queryParams};
      params.pageNo = undefined;
      params.pageSize = undefined;
      this.$modal.confirm('是否确认导出所有文档列表数据项?').then(() => {
        this.exportLoading = true;
        return exportDocumentExcel(params);
      }).then(response => {
        this.$download.excel(response, '文档列表.xls');
        this.exportLoading = false;
      }).catch(() => {});
    },

    // 判断文件类型和大小
    beforeAvatarUpload(file) {
      console.log(file,"file")
      // const isJPG = file.type === 'image/jpeg';
      const isLt16M = file.size / 1024 / 1024 < 16;
      const isLength5 = this.fileList.filter(item=>item.status != 2).length < 5;
      const isSame = this.fileList.filter(item=>item.status != 2).some(item => item.name === file.name);

      // if (!isJPG) {
      //   this.$message.error('上传头像图片只能是 JPG 格式!');
      // }
      if (!isLt16M) {
        this.$message.error('上传文件大小不能超过 16MB!');
      }
      if (!isLength5) {
        this.$message.error('上传文件数量不能超过 5个');
      }
      if (isSame) {
        this.$message.error('不能上传相同的文件');
      }

      this.fileList = _.cloneDeep(this.fileList)
      return isLt16M && isLength5 && !isSame;
    },

    // 文件状态改变时的钩子，添加文件、上传成功和上传失败时都会被调用,function(file, fileList)
    handleChange(file, fileList) {
      if (!this.beforeAvatarUpload(file)) return
      file.status = 1;
      this.fileList.push(file)
    },
    // 删除文件之前的钩子，参数为上传的文件和文件列表，若返回 false 或者返回 Promise 且被 reject，则停止删除。function(file, fileList)
    handleRemove(file, fileList) {
      this.fileList.forEach(item=>{
        if (item.uid === file.uid){
          item.status = 2
        }
      });
      if (this.form.id != null) {
        this.form.attachPathList.forEach(item=>{
          if (item.id === file.id){
            item.status = 2
          }
        })
      }
    },

    // 将文档附件上传到服务器
    async submitUpload() {
      //判断是否有文件再上传
      // if (this.fileList.length === 0) {
      //   this.dialogLoading = false;
      //   this.$message.warning('请选取文件后再上传')
      //   return false;
      // }
      // 下面的代码将创建一个空的FormData对象:
      const formData = new FormData()

      // 有新增的才调用导入文档附件接口
      if (this.fileList.some(item=>item.status===1)){

        this.fileList.forEach((file) => {
          if (file.status===1){
            // 你可以使用FormData.append来添加键/值对到表单里面；
            formData.append('files', file.raw)
          }
        })

        // 添加自定义参数，不传可删除
        // formData.append('parentId', '49')

        // 导入文档附件
        try {
          const formQuery = {}
          if (this.form.id != null && this.form.documentType != null && this.form.documentType != ""  && this.form.documentType != undefined){
            formQuery.type = this.form.documentType
          }else {
            formQuery.type = 0
          }
          const res = await importingDocumentAttachments(formQuery,formData)
          this.$message.success('上传成功');
          this.documentAttachments.push(...res.data);
          this.documentAttachments.forEach(item => item.status = 1);
        }catch (e){
          this.dialogLoading = false;
          this.$message.error('上传失败');
          return false
        }
      }

      return true;
    },

    // 下载文档附件
    async downloadDocument(attachPath){
      try {
        const res = await downloadDocumentAttachments({fileName:attachPath.fileName})
      }catch (e) {
        this.$message.error(e);
      }
    },
    // 调用 drag-table.vue 的方法
    callDragTableMethod(e) {
      this.$refs.dragTableRef.displayChange(e);
    },
    // 初始化显示项
    handleDisplay() {
      this.$refs.dragTableRef.handleDisplay();
    }
  }
};
